阶段2: 数据库设计
-- 论文主表
CREATE TABLE papers (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    arxiv_id VARCHAR(100) UNIQUE NOT NULL,
    doi VARCHAR(200) UNIQUE,
    title TEXT NOT NULL,
    title_chinese TEXT,
    authors TEXT NOT NULL,
    year INTEGER,
    abstract TEXT,
    abstract_chinese TEXT,
    pdf_path TEXT,
    arxiv_url TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    parent_paper_id INTEGER,  -- 用于关联主论文和相关论文
    FOREIGN KEY (parent_paper_id) REFERENCES papers(id)
);

-- 参考文献表
CREATE TABLE references (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    paper_id INTEGER NOT NULL,
    ref_arxiv_id VARCHAR(100),
    ref_title TEXT,
    FOREIGN KEY (paper_id) REFERENCES papers(id)
);

-- 支持多表查询的视图
CREATE VIEW paper_details AS
SELECT 
    p.*,
    GROUP_CONCAT(r.ref_arxiv_id) as reference_ids
FROM papers p
LEFT JOIN references r ON p.id = r.paper_id
GROUP BY p.id;

技术难点与解决方案
难点1: 多种输入格式处理
解决方案: 编写统一的解析器，支持多种arXiv标识格式
难点2: 参考文献提取
解决方案: 从arXiv API获取，如果API不提供，从PDF中尝试提取
难点3: 翻译API稳定性
解决方案:
首选LibreTranslate API
备用：使用Google Translate的免费配额
最后备用：使用本地翻译库（质量较低）
难点4: 相关论文获取
解决方案: 使用arXiv官方API的"related"功能，避免爬虫问题
难点5: 大文件上传处理
解决方案: 配置Flask的文件大小限制，分块上传，进度提示
